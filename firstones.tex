\luadirect{
local function makenode(a, xpos, ypos, fillcolor,filltext)
    tp("\\node at ("..xpos*xscale..","..-ypos*yscale..")[fill = "..fillcolor.."!20,minimum width=1.3cm,minimum height = .9cm]{\\textsubscript{ "..a['number'].."}".. filltext.."};")  
end

xscale = 1.5
yscale = 1
tp = tex.print

local tio = texio.write_nl

function makePTE(find,findcolor)    
	tp("\\node at (9,-1)[fill = red,minimum width=1.3cm,minimum height = .9cm]{"..find.."};")  
	for _,a in ipairs(atoms) do
	  if a["number"] then
	    x = a["number"]
	    tio(x .. a["short"])
	    
	    if (x < 119) then xpos = x-86-14 ypos = 7 end	    
	    if (x < 104) then xpos = x-86 ypos = 9.2 end	    
	    if (x < 89) then xpos = x-86 ypos = 7 end
	    if (x < 87) then xpos = x-68 ypos = 6 end
	    if (x < 72) then xpos = x-54 ypos = 8.2 end
	    if (x < 57) then xpos = x-54 ypos = 6 end
	    if (x < 55) then xpos = x-36 ypos = 5 end
	    if (x < 37) then xpos = x-18 ypos = 4 end
	    if (x < 19) then xpos = x-10+10 ypos = 3 end
	    if (x < 13) then xpos = x-10 ypos = 3 end
	    if (x < 11) then xpos = x-2+10 ypos = 2 end
	    if (x < 5) then xpos = x-2 ypos = 2 end
	    if (x < 3) then xpos = 17*(x-1)+1 ypos = 1 end
	
	    fillcolor = "blue"
	    filltext = a["short"]
	    if a[find] then
	      fillcolor = findcolor
	      filltext = "\\href{"..a[find].."}{"..a["short"].."}"
	    end
	    makenode(a, xpos,ypos,fillcolor,filltext)
	  end
	end
end
}
